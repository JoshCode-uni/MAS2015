% GOAL MANAGEMENT
		
			%% See goals %%
			
			%% If I see A flag on the floor, get flag. (Priority. breaks one goal policy)
			if bel( flag(Team,_, Position), 
					flagState(Team,dropped)) then adopt( goTo(Position) ).
			
			%% If I have flag, bring back to base.
			if not( goal( goTo(_) ) ), bel( self(_,_, Team),
					flag(not(Team), HolderUnrealID, Position), 
					flagState(not(Team), held), self(HolderUnrealID, _, _), 
					base(Team, Destination) ) then adopt( goTo(Destination) ).
			
			%% If see armor and wantArmor(Type), then make armor goal.
			if not( goal( goTo(_) ) ), bel( item(Destination,armor,_,_),wantArmor(Type), not(goTo(Destination))) then adopt( goTo(Destination) ).
					
			%% If I see pickup and I am not under effect of one, make pickup new goal. 
			if not( goal( goTo(_) ) ), bel( item(Destination,Type,_,_), powerup(Type,Time), Time@=<1, not(at(Destination))) then adopt( goTo(Destination) ).
			
			%% Want goals %%
			
			%% If I have a weapon and ammo count is 5 or lower, make ammo new goal. (pickup is used because this both avoids running dry in combat and adds movement in combat which can be beneficial)
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,shock_rifle), not(goTo(Destination)), weapon(shock_rifle,Ammo,_), Ammo@=<10 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,bio_rifle), not(goTo(Destination)), weapon(bio_rifle,Ammo,_), Ammo@=<20 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,link_gun), not(goTo(Destination)), weapon(link_gun,Ammo,_), Ammo@=<40 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,stinger_minigun), not(goTo(Destination)), weapon(stinger_minigun,Ammo,_), Ammo@=<40 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,flak_cannon), not(goTo(Destination)), weapon(flak_cannon,Ammo,_), Ammo@=<8 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,rocket_launcher), not(goTo(Destination)), weapon(rocket_launcher,Ammo,_), Ammo@=<6 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,sniper_rifle), not(goTo(Destination)), weapon(sniper_rifle,Ammo,_), Ammo@=<8 ) then adopt( goTo(Destination) ).
			if not( goal( goTo(_) ) ), bel( pickup(Destination,ammo,avril), not(goTo(Destination)), weapon(avril,Ammo,_), Ammo@=<5 ) then adopt( goTo(Destination) ).
			
			%% If health is lower then 100, make pack new goal.
			if not( goal( goTo(_) ) ), bel( pickup(Destination,health,_), status(Health,_), Health@<100, not(goTo(Destination))) then adopt( goTo(Destination) ).
			
			%% If I have no goals, go to enemy base. (Placeholder)
			if not( goal( goTo(_) ) ), bel( self(_, _, Team), status(Health,_), base(OtherTeam, Destination), not(Team = OtherTeam), not(goTo(Destination))) then adopt( goTo(Destination)).
